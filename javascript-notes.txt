var Sum = function(){

  function sum () {
      var result = 0;
      for (var i = 0; i < arguments.length; i++) {
          result += arguments[i];
      }
      return result;
  }
  
  if (arguments.length == 0) {
    //no arguments, return function
    return sum;
  }else{
    //Arguments 
    return sum.apply(null, arguments);
  }
};
